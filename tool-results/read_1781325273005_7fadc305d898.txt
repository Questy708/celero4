   255→export function Infra() {
   256→  const [activeSection, setActiveSection] = useState(0);
   257→
   258→  useEffect(() => {
   259→    const observer = new IntersectionObserver(
   260→      (entries) => {
   261→        entries.forEach((entry) => {
   262→          if (entry.isIntersecting) {
   263→            const idx = sectionIds.indexOf(entry.target.id);
   264→            if (idx !== -1) setActiveSection(idx);
   265→          }
   266→        });
   267→      },
   268→      { threshold: 0.3, rootMargin: "-10% 0px -10% 0px" }
   269→    );
   270→    sectionIds.forEach((id) => {
   271→      const el = document.getElementById(id);
   272→      if (el) observer.observe(el);
   273→    });
   274→    return () => observer.disconnect();
   275→  }, []);
   276→
   277→  return (
   278→    <div className="bg-white text-[#111]">
   279→      <StickySideNav activeIndex={activeSection} />
   280→      <HeroSection />
   281→      <CaseSection />
   282→      <OutpostModelSection />
   283→      <LevelsSection />
   284→      <CampusRingSection />
   285→      <CoreTechSection />
   286→      <FlowSection />
   287→      <MoatSection />
   288→      <HubsSection />
   289→      <CivilizationSection />
   290→      <ReviewSection />
   291→    </div>
   292→  );
   293→}
